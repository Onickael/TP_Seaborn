Informations sur les données :

Les cartes de credit Scoring (évaluation des risques) sont une méthode courante de contrôle des risques dans le secteur financier. Il utilise les informations personnelles et les données soumises par les demandeurs de carte de crédit pour prédire la probabilité de futurs défauts de paiement et d'emprunts de carte de crédit. La banque est en mesure de décider d'émettre une carte de crédit au demandeur. Les Credit Scores peuvent quantifier objectivement l'ampleur du risque.

Le diagramme de dispersion tente de montrer la relation entre les jours travaillés et l'âge de la personne (DAYS_BIRTH) pour les personnes qui n'étaient pas sans emploi. import numpy as np
import pandas as pd
import seaborn as sns

import matplotlib.pyplot as plt

list_url = ['./csv/application_record-1.csv',
            './csv/application_record-2.csv',
            './csv/application_record-3.csv']

df = pd.concat([pd.read_csv(f) for f in list_url],ignore_index=True)

birth = df['DAYS_BIRTH'].abs()

doubled_ages = []
for age in birth:
    doubled_ages.append(age)
    doubled_ages.append(age)


doubled_ages_years = [age / 365.25 for age in doubled_ages]
doubled_ages_years_cleaned = [age for age in doubled_ages_years if not np.isnan(age)]

sns.displot(x=doubled_ages_years_cleaned)
plt.xlabel('Age en année')
plt.ylabel('Nombre')

plt.savefig('Age en année.png')

import numpy as np
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

list_url = ['./csv/application_record-1.csv',
            './csv/application_record-2.csv',
            './csv/application_record-3.csv']

df = pd.concat([pd.read_csv(f) for f in list_url], ignore_index=True)

df_sorted = df.sort_values(by='AMT_INCOME_TOTAL')

half_data = df_sorted.iloc[:len(df_sorted)//2]

sns.boxplot(x='NAME_FAMILY_STATUS', y='AMT_INCOME_TOTAL', hue='FLAG_OWN_REALTY', data=half_data)

plt.title('Distribution du revenu total pour la moitié inférieure des salariés')
plt.xlabel('Situation familiale')
plt.ylabel('Revenu total')
plt.savefig('situation familiale.png')

import numpy as np
import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt

list_url = ['./csv/application_record-1.csv',
            './csv/application_record-2.csv',
            './csv/application_record-3.csv']

df = pd.concat([pd.read_csv(f) for f in list_url], ignore_index=True)

df_numeric = df.drop(columns=['FLAG_MOBIL']).select_dtypes(include=[np.number])

corr = df_numeric.corr()

sns.heatmap(corr, annot=True, cmap='coolwarm', fmt='.2f', linewidths=0.5)
plt.savefig('correlation.png')
